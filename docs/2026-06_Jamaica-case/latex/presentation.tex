% --- Slide: key takeaways (links to the supporting tables/figures) ---
\begin{frame}{Key Takeaways}
    \small
    \begin{itemize}
        \setlength{\itemsep}{0.6em}
        \item \textbf{Jamaica's spending inefficiency is in human capital, not infrastructure.} Its infrastructure efficiency gap is small (0.17), while health and education gaps are large ($\approx$0.36).
            {\footnotesize\par\hfill\hyperref[fig:efficiencyBandsJAM]{Fig.~\ref*{fig:efficiencyBandsJAM}}, \hyperref[slide:calib]{calibration table}}
        \item \textbf{Reallocating 1\% of GDP to investment raises output, with multipliers that build over time.} Under Jamaica's calibration the infrastructure multiplier reaches $\approx$3.8 by 2050.
            {\footnotesize\par\hfill\hyperref[slide:mult]{multiplier table}}
        \item \textbf{Jamaica's own calibration amplifies that multiplier.} Low trend growth (1.6\%), high debt (75\%) and low public investment (3\% of GDP) raise the return to public capital.
            {\footnotesize\par\hfill\hyperref[slide:params]{parameters}}
        \item \textbf{The largest gains pair reallocation with closing efficiency gaps --- for Jamaica that favours human capital.} By 2050 reallocation alone adds $\approx$5--6\% of GDP; closing the efficiency gaps lifts human capital to $\approx$10\% and infrastructure to $\approx$9\%.
            {\footnotesize\par\hfill\hyperref[fig:combinedEM]{Figs~\ref*{fig:combinedEM}--\ref*{fig:combinedJAM}}}
    \end{itemize}
\end{frame}

    \input{multipliersTable}
    \end{center}

    \vspace{0.8em}
    {\fontsize{6pt}{7.5pt}\selectfont
        Sources: IMF staff estimates (model simulations). \\
        Notes: Cumulative output multiplier over $N$ years ($4N$ quarters):
        $\; m_N = \sum_{t=1}^{4N}(y_t-y_0)\big/\sum_{t=1}^{4N}\Delta G_t$,
        where $t$ indexes quarters from the first period the shock is active, $y_0$ is the pre-shock steady state, and $\Delta G_t = (I_{g,t}-I_g^{*}) + (C_{ge,t}-C_{ge}^{*}) + (C_{grd,t}-C_{grd}^{*})$ is the spending increase, for a permanent rise of 1 percent of GDP in the category funded by cutting public consumption. Multipliers build over time as public capital accumulates. Jamaica differs from EMDEs through its lower trend growth, higher debt and smaller public-investment share (Bank of Jamaica calibration), which lift its infrastructure multiplier in particular. \\
    }
\end{frame}
